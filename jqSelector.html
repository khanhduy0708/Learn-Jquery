Bộ chọn Selector trong jQuery: 

                // selector tên thẻ html
                $("div").css("background", "orange");

                // selector #id
                $("#p1").css("background", "green");

                // selector class .class
                $(".class1").css("background", "black");

                // selector thuộc tính
                $("input[name='username']").css("padding", "40px");

                // selector lồng selector
                $(".row .col").css("padding", "40px");
                
                // selector con trực tiếp của selector
                $(".container > .row").css("padding", "40px");
